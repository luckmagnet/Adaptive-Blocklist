name: Convert Hosts to Adblock

on:
  push:
    paths:
      - '**/conversion.yaml'
      - '**/convert_hosts_to_adblock.py'
      - '**/hosts'
      - '**/hosts.txt'
  workflow_dispatch:

jobs:
  convert:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - name: Set up Python
        uses: actions/setup-python@v4
        with:
          python-version: '3.11'

      - name: Install dependencies
        run: python -m pip install --upgrade pip pyyaml

      - name: Run converter
        shell: bash
        run: |
          set -euo pipefail
          # prefer hosts.txt at repo root
          if [ -f hosts.txt ]; then
            HOSTS_FILE=hosts.txt
          elif [ -f hosts ]; then
            HOSTS_FILE=hosts
          else
            echo "ERROR: hosts file not found (checked 'hosts.txt' and 'hosts')" >&2
            exit 2
          fi
          CFG_FILE=conversion.yaml
          if [ ! -f "$CFG_FILE" ]; then
            echo "ERROR: config file not found at '$CFG_FILE'" >&2
            exit 2
          fi
          python -u convert_hosts_to_adblock.py "$HOSTS_FILE" "$CFG_FILE" > adblock.txt

      - name: Upload adblock as artifact
        uses: actions/upload-artifact@v4
        with:
          name: adblock-list
          path: adblock.txt
